111        """Add a node to the tree following the defined type constrains"""

119    def _should_add(self, pos: int, new_node: TreeNode) -> bool:
120        """
121        Recursively find the parent function of the new node and check it
122        matches by type to the expected parameter. (pos is a size-1 list
123        so as to pass "by reference" on successive recursive calls).
124        """
125        if (
126            pos[0] == self.size()
127        ):  # reached the place the new node should be placed
128            return True
129
130        node = self.tree[pos[0]]
131        res = None
132        if isinstance(node, FunctionNode):
133            func_types = get_func_types(node.function)
134            for i in range(node.n_args):
135                pos[0] += 1
136                res = self._should_add(pos, node)
137                if res:
138                    return func_types[i] == new_node.node_type
139                elif res is not None:
140                    return res
141        return res

157    def _depth(self, pos, d) -> int:

160        on successive recursive calls).
161        """
162

164
165        depths = []
166        if isinstance(node, FunctionNode):
167            for _ in range(node.n_args):

194    def random_terminal(
195        self,
196        node_type: Optional[type] = None
197    ) -> Optional[TerminalNode]:
198        """Select a random terminal, including constants from ERC range"""
199        relevant_terminals = [
200            term
201            for term, term_type in self.terminal_set.items()
202            if term_type == node_type
203        ]
204
205        if self.erc_range is not None and (
206            node_type is None or issubclass(node_type, Number)
207        ):
208            relevant_terminals.append(
209                random.uniform(*self.erc_range)
210                if type(self.erc_range[0]) is float
211                else random.randint(*self.erc_range)

214        # Return None in case there are no terminals of the given type
215        if not relevant_terminals:
216            return None

221        # and int/float in typed case

227    def execute(self, *args, **kwargs) -> object:
228        """
229        Execute the program (tree).
230        Input is a numpy array or keyword arguments (but not both).
231
232        Parameters
233        ----------
234        args : arguments
235            A numpy array.
236
237        kwargs : keyword arguments
238            Input to program, including every variable
239            in the terminal set as a keyword argument.
240            For example, if `terminal_set=['x', 'y', 'z']`
241            then call `execute(x=..., y=..., z=...)`.
242
243        Returns
244        -------
245        object
246            Result of tree execution.
247        """
248
249        if not self.tree:
250            raise ValueError("Tree is empty, cannot execute.")
251
252        reshape = False
253        if args:  # numpy array -- convert to kwargs
254            try:
255                X = args[0]
256                kwargs = generate_args(X)
257                reshape = True
258            except Exception:
259                raise ValueError(
260                    f"Bad argument to tree.execute, "
261                    f"must be np array or kwargs: {args}"
262                )
263
264        kw = list(kwargs.keys())
265
266        bad_vars = [item for item in kw if item not in self.terminal_set]
267        if len(bad_vars) > 0:
268            raise ValueError(
269                f"received variable arguments not in terminal set: {bad_vars}"
270            )
271
272        missing_vars = [item for item in self.terminal_set if item not in kw]
273        if len(missing_vars) > 0:
274            raise ValueError(
275                f"Missing variable terminals as execute kwargs: {missing_vars}"
276            )
277
278        res = self._execute([0], **kwargs)
279
280        if reshape and (isinstance(res, Number) or res.shape == np.shape(0)):
281            # sometimes a tree degenrates to a scalar value
282            res = np.full_like(X[:, 0], res)
283        return res

285    def _execute(self, pos, **kwargs):
286        """
287        Recursively execute the tree by traversing it in a depth-first order
288        (pos is a size-1 list so as to pass "by reference"
289        on successive recursive calls).
290        """
291
292        node = self.tree[pos[0]]
293
294        if isinstance(node, FunctionNode):
295            arglist = []
296            for _ in range(node.n_args):
297                pos[0] += 1
298                res = self._execute(pos, **kwargs)
299                arglist.append(res)
300            return node.function(*arglist)
301        else:  # terminal
302            if node.value in kwargs:
303                # terminal is a variable, return its value
304                return kwargs[node.value]
305            else:  # terminal is a constant
306                return node.value

331    def random_subtree(
332        self,
333        node_type: Optional[type] = None
334    ) -> Optional[List[TreeNode]]:
335        relevant_nodes = self.filter_tree(
336            lambda node: node.node_type is None  # untyped case
337            or (

339            )  # typed case with first invocation

342            )  # typed case with subsequent invocations

346        subtree_root = random.choice(relevant_nodes)

350    def _get_subtree_by_root(self, subtree_root: TreeNode) -> List[TreeNode]:
351        start_i = self.tree.index(subtree_root)
352
353        if start_i == 0:
354            return self.tree
355
356        end_i = self._find_subtree_end([start_i])
357        return self.tree[start_i: end_i + 1]

359    def replace_subtree(
360        self, old_subtree: List[TreeNode], new_subtree: List[TreeNode]
361    ) -> None:
362        """
363        Replace the subtree starting at `index` with `subtree`
364
365        Parameters
366        ----------
367        subtree - new subtree to replace the existing subtree in the tree
368
369        Returns
370        -------
371        None
372        """
373        start_i = self.tree.index(old_subtree[0])
374        end_i = start_i + len(old_subtree)
375        left_part = self.tree[:start_i]
376        right_part = self.tree[end_i:]
377        self.tree = left_part + new_subtree + right_part

379    def _find_subtree_end(self, pos):
380        """find index of final node of subtree that starts at `pos`
381        (pos is a size-1 list so as to pass "by reference" on successive recursive calls).
382        """
383
384        node = self.tree[pos[0]]
385
386        if (
387            isinstance(node, FunctionNode)
388            and node.function in self.function_set
389        ):
390            for _ in range(node.n_args):
391                pos[0] += 1
392                self._find_subtree_end(pos)
393
394        return pos[0]

396    def _handle_input_types(
397        self,
398        function_set: List[Callable],
399        terminal_set: Union[Dict[Any, type], List[str]],
400        root_type: type
401    ):
402        if function_set is None:
403            raise ValueError("Function set must be provided.")
404
405        if terminal_set is None:
406            raise ValueError("Terminal set must be provided.")
407
408        for t in function_set:
409            if not isinstance(t, Callable):
410                raise ValueError(
411                    f"Functions must be Callble, but {t} is of type {type(t)}"
412                )
413
414        if not isinstance(terminal_set, (list, dict)):
415            raise ValueError(
416                "Terminal set must be a list or a dictionary, "
417                f"got {type(terminal_set)}."
418            )
419
420        # untyped case - convert to dictionary of Nones.
421        if isinstance(terminal_set, list):
422            # check if any function has type hints
423            if any(f.__annotations__ for f in function_set):
424                raise ValueError(
425                    "Detected typed function with untyped terminal set. \
426                    Please provide a dictionary with types for terminals."
427                )
428
429            terminal_set = {t: None for t in terminal_set}
430            return function_set, terminal_set
431
432        # typed case - check every value is a type
433        for v in terminal_set.values():
434            if not isinstance(v, type):
435                raise ValueError(
436                    "Values in terminal set dictionary must be types, "
437                    f"but {v} is of type {type(v)}."
438                )
439
440        function_return_types = {
441            get_return_type(f) for f in function_set
442        }
443        if root_type not in function_return_types:
444            raise ValueError(
445                f"Detected a mismatch between root_type ({root_type}) "
446                f"and function set ({function_set}).\n"
447                f"Root type must be the return type of at least one function."
448            )
449        if self.erc_type is not None and self.erc_type not in function_return_types:
450            raise ValueError(
451                f"Detected a mismatch between ERC type ({self.erc_type}) "
452                f"and function set ({function_set}).\n"
453                f"ERC range should not be defined if there are no numeric functions."
454            )
455
456        # check terminals and functions type intersection
457        function_arg_types = {
458            t for func in function_set for t in get_func_types(func)[:-1]
459        }
460        terminal_types = set(terminal_set.values())
461
462        if self.erc_type:
463            terminal_types.add(self.erc_type)
464
465        if function_arg_types != terminal_types:
466            raise ValueError(
467                f"Function args type hints ({function_arg_types}) "
468                f"must match terminal types ({terminal_types})."
469            )
470
471        return function_set, terminal_set

473    def __str__(self) -> str:
474        """
475        Return a simple textual representation of the tree.
476
477        Returns
478        -------
479        str
480            String representation of the tree.
481
482        Examples
483        --------
484        untyped case:
485        >>> ut = Tree(terminal_set=[x, y, z], function_set=[f_add])
486        >>> ut.tree = [FunctionNode(f_add), TerminalNode(1), TerminalNode(2)]
487        >>> tree.show()
488        def func_0(x, y, z):
489            return f_add(x, y)
490
491        typed case:
492        >>> tt = Tree(terminal_set={x: float, y: float},
493        ...           function_set=[typed_add])
494        >>> tt.tree = [FunctionNode(typed_add),
495        ...            TerminalNode(1.0, float),
496        ...            TerminalNode(2.0, float)]
497        >>> str(tree)
498        def func_0(x: float, y: float) -> float:
499            return typed_add(1.0, 2.0)
500        """
501        args = (
502            list(self.terminal_set.keys())
503            if None in self.terminal_set.values()
504            else [f"{k}: {v.__name__}" for k, v in self.terminal_set.items()]
505        )
506        ret_type_str = (
507            ""
508            if None in self.terminal_set.values()
509            else f" -> {self.root.node_type.__name__}"
510        )
511
512        result = [
513            f"def func_{str(self.id)}({', '.join(args)}){ret_type_str}:\n  return "
514        ]
515        self._str_rec("", [0], result)
516        return "".join(result)
517
518    def _str_rec(self, prefix: str, pos: List[int], result: str) -> str:
519        """Recursively produce a simple textual printout of the tree
520        (pos is a size-1 list so as to pass "by reference" on successive
521         recursive calls).
522        """
523
524        node = self.tree[pos[0]]
525        if isinstance(node, FunctionNode):
526            result.append(f"{prefix}{str(node)}(\n")
527            prefix = prefix + "  " if pos[0] == 0 else prefix
528            for i in range(node.n_args):
529                pos[0] += 1
530                self._str_rec(prefix + "  ", pos, result)
531                if i < node.n_args - 1:
532                    result.append(",\n")
533                elif 0 < i < node.n_args:
534                    result.append("\n")
535            result.append(prefix + ")")
536        else:  # terminal
537            result.append(f"{prefix}{str(node)}")

232

234        ----------
235        args : arguments
236            A numpy array.
237
238        kwargs : keyword arguments
239            Input to program, including every variable
240            in the terminal set as a keyword argument.
241            For example, if `terminal_set=['x', 'y', 'z']`
242            then call `execute(x=..., y=..., z=...)`.

245        -------
246        object
247            Result of tree execution.

286    def _execute(self, pos, **kwargs):
287        """
288        Recursively execute the tree by traversing it in a depth-first order
289        (pos is a size-1 list so as to pass "by reference"
290        on successive recursive calls).
291        """
292
293        node = self.tree[pos[0]]
294
295        if isinstance(node, FunctionNode):
296            arglist = []
297            for _ in range(node.n_args):
298                pos[0] += 1
299                res = self._execute(pos, **kwargs)
300                arglist.append(res)
301            return node.function(*arglist)
302        else:  # terminal
303            if node.value in kwargs:
304                # terminal is a variable, return its value
305                return kwargs[node.value]
306            else:  # terminal is a constant
307                return node.value

380    def _find_subtree_end(self, pos):
381        """find index of final node of subtree that starts at `pos`
382        (pos is a size-1 list so as to pass "by reference" on successive recursive calls).
383        """

388            isinstance(node, FunctionNode)

397    def _handle_input_types(
398        self,
399        function_set: List[Callable],
400        terminal_set: Union[Dict[Any, type], List[str]],
401        root_type: type
402    ):
403        if function_set is None:
404            raise ValueError("Function set must be provided.")
405
406        if terminal_set is None:
407            raise ValueError("Terminal set must be provided.")
408
409        for t in function_set:
410            if not isinstance(t, Callable):
411                raise ValueError(
412                    f"Functions must be Callble, but {t} is of type {type(t)}"
413                )
414
415        if not isinstance(terminal_set, (list, dict)):
416            raise ValueError(
417                "Terminal set must be a list or a dictionary, "
418                f"got {type(terminal_set)}."
419            )
420
421        # untyped case - convert to dictionary of Nones.
422        if isinstance(terminal_set, list):
423            # check if any function has type hints
424            if any(f.__annotations__ for f in function_set):
425                raise ValueError(
426                    "Detected typed function with untyped terminal set. \
427                    Please provide a dictionary with types for terminals."
428                )
429
430            terminal_set = {t: None for t in terminal_set}
431            return function_set, terminal_set
432
433        # typed case - check every value is a type
434        for v in terminal_set.values():
435            if not isinstance(v, type):
436                raise ValueError(
437                    "Values in terminal set dictionary must be types, "
438                    f"but {v} is of type {type(v)}."
439                )
440
441        function_return_types = {
442            get_return_type(f) for f in function_set
443        }
444        if root_type not in function_return_types:
445            raise ValueError(
446                f"Detected a mismatch between root_type ({root_type}) "
447                f"and function set ({function_set}).\n"
448                f"Root type must be the return type of at least one function."
449            )
450        if self.erc_type is not None and self.erc_type not in function_return_types:
451            raise ValueError(
452                f"Detected a mismatch between ERC type ({self.erc_type}) "
453                f"and function set ({function_set}).\n"
454                f"ERC range should not be defined if there are no numeric functions."
455            )
456
457        # check terminals and functions type intersection
458        function_arg_types = {
459            t for func in function_set for t in get_func_types(func)[:-1]
460        }
461        terminal_types = set(terminal_set.values())
462
463        if self.erc_type:
464            terminal_types.add(self.erc_type)
465
466        if function_arg_types != terminal_types:
467            raise ValueError(
468                f"Function args type hints ({function_arg_types}) "
469                f"must match terminal types ({terminal_types})."
470            )
471
472        return function_set, terminal_set

474    def __str__(self) -> str:
475        """
476        Return a simple textual representation of the tree.
477
478        Returns
479        -------
480        str
481            String representation of the tree.
482
483        Examples
484        --------
485        untyped case:
486        >>> ut = Tree(terminal_set=[x, y, z], function_set=[f_add])
487        >>> ut.tree = [FunctionNode(f_add), TerminalNode(1), TerminalNode(2)]
488        >>> tree.show()
489        def func_0(x, y, z):
490            return f_add(x, y)
491
492        typed case:
493        >>> tt = Tree(terminal_set={x: float, y: float},
494        ...           function_set=[typed_add])
495        >>> tt.tree = [FunctionNode(typed_add),
496        ...            TerminalNode(1.0, float),
497        ...            TerminalNode(2.0, float)]
498        >>> str(tree)
499        def func_0(x: float, y: float) -> float:
500            return typed_add(1.0, 2.0)
501        """
502        args = (
503            list(self.terminal_set.keys())
504            if None in self.terminal_set.values()
505            else [f"{k}: {v.__name__}" for k, v in self.terminal_set.items()]
506        )
507        ret_type_str = (
508            ""
509            if None in self.terminal_set.values()
510            else f" -> {self.root.node_type.__name__}"
511        )
512
513        result = [
514            f"def func_{str(self.id)}({', '.join(args)}){ret_type_str}:\n  return "
515        ]
516        self._str_rec("", [0], result)
517        return "".join(result)

519    def _str_rec(self, prefix: str, pos: List[int], result: str) -> str:
520        """Recursively produce a simple textual printout of the tree
521        (pos is a size-1 list so as to pass "by reference" on successive
522         recursive calls).
523        """
524
525        node = self.tree[pos[0]]
526        if isinstance(node, FunctionNode):
527            result.append(f"{prefix}{str(node)}(\n")
528            prefix = prefix + "  " if pos[0] == 0 else prefix
529            for i in range(node.n_args):
530                pos[0] += 1
531                self._str_rec(prefix + "  ", pos, result)
532                if i < node.n_args - 1:
533                    result.append(",\n")
534                elif 0 < i < node.n_args:
535                    result.append("\n")
536            result.append(prefix + ")")
537        else:  # terminal
538            result.append(f"{prefix}{str(node)}")
