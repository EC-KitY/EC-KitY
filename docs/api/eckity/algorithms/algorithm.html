 29class Algorithm(Operator, ABC):
 30    """
 31    Evolutionary algorithm to be executed.
 32
 33    Abstract Algorithm that can be extended to concrete algorithms,
 34    such as SimpleEvolution, Coevolution etc.
 35
 36    Parameters
 37    ----------
 38    population: Population
 39        The population to be evolved.
 40        Consists of several sub-populations.
 41        ref: https://api.eckity.org/eckity/population.html
 42
 43    statistics: Statistics or list of Statistics, default=None
 44        Provide statistics on the population during the evolution.
 45        ref: https://api.eckity.org/eckity/statistics.html
 46
 47    breeder: Breeder, default=SimpleBreeder()
 48        Responsible for applying selection and operator sequence on individuals
 49        in each generation. Applies on one sub-population in simple case.
 50        ref: https://api.eckity.org/eckity/breeders.html
 51
 52    population_evaluator: PopulationEvaluator,
 53                          default=SimplePopulationEvaluator()
 54        Evaluates individual fitness scores concurrently and returns the best
 55        individual of each subpopulation (one individual in simple case).
 56        ref: https://api.eckity.org/eckity/evaluators.html
 57
 58    max_generation: int, default=100
 59        Maximal number of generations to run the evolutionary process.
 60        Note the evolution could end before reaching max_generation,
 61        depends on the termination checker.
 62        Note there are up to `max_generation + 1` fitness calculations,
 63        but only up to `max_generation` selections
 64
 65    events: dict(str, dict(object, function)), default=None
 66        dict of events, each event holds a dict (subscriber, callback).
 67
 68    event_names: list of strings, default=None
 69        Names of events to publish during the evolution.
 70
 71    termination_checker: TerminationChecker or a list of TerminationCheckers,
 72                          default=None
 73        Checks if the algorithm should terminate early.
 74        ref: https://api.eckity.org/eckity/termination_checkers.html
 75
 76    max_workers: int, default=None
 77        Maximal number of worker nodes for the Executor object
 78        that evaluates the fitness of the individuals.
 79        ref: https://docs.python.org/3/library/concurrent.futures.html
 80
 81    random_generator: RNG, default=RNG()
 82        Random Number Generator.
 83
 84    random_seed: int, default=current system time
 85        Random seed for deterministic experiment.
 86
 87    generation_seed: int, default=None
 88        Current generation seed.
 89        Useful for resuming a previously paused experiment.
 90
 91    generation_num: int, default=0
 92        Current generation number
 93
 94    Attributes
 95    ----------
 96    final_generation_: int
 97        The generation in which the evolution ended.
 98    """

172    def evolve(self) -> None:
173        """
174        Performs the evolutionary run by initializing the random seed,
175        creating the population, performing the evolutionary loop
176        and finally finishing the evolution process
177        """
178        self.initialize()
179
180        if self.should_terminate(
181            self.population, self.best_of_run_, self.generation_num
182        ):
183            self.final_generation_ = 0
184            self.publish("after_generation")
185        else:
186            self.evolve_main_loop()
187
188        self.publish("evolution_finished")
189        self.finish()

213        """
214        Initialize seed, Executor and relevant operators

228    def _validate_population_type(self, population: Any) -> None:
229        # Assert valid population input
230        if population is None:
231            raise ValueError("Population cannot be None")
232
233        if isinstance(population, Population):
234            self.population = population
235        elif isinstance(population, Subpopulation):
236            self.population = Population([population])
237        elif isinstance(population, list):
238            if len(population) == 0:
239                raise ValueError("Population cannot be empty")
240            for sub_pop in population:
241                if not isinstance(sub_pop, Subpopulation):
242                    raise ValueError(
243                        "Detected a non-Subpopulation "
244                        "instance as an element in Population"
245                    )
246            self.population = Population(population)
247        else:
248            raise ValueError(
249                "Parameter population must be either a Population, "
250                "a Subpopulation or a list of Subpopulations. "
251                "Received population with unexpected type of",
252                type(population),
253            )

255    def _validate_statistics_type(self, statistics: Any) -> None:
256        # Assert valid statistics input
257        if isinstance(statistics, Statistics):
258            self.statistics = [statistics]
259        elif isinstance(statistics, list):
260            for stat in statistics:
261                if not isinstance(stat, Statistics):
262                    raise ValueError(
263                        "Expected a Statistics instance as an element"
264                        " in Statistics list, but received",
265                        type(stat),
266                    )
267            self.statistics = statistics
268        else:
269            raise ValueError(
270                "Parameter statistics must be either a subclass of Statistics"
271                " or a list of subclasses of Statistics.\n"
272                "received statistics with unexpected type of",
273                type(statistics),
274            )

276    def evolve_main_loop(self) -> None:
277        """
278        Performs the evolutionary main loop
279        """
280        # there was already "preprocessing" generation created - gen #0
281        # now create another self.max_generation generations, starting gen #1
282        for gen in range(1, self.max_generation + 1):
283            self.generation_num = gen
284            self.update_gen(gen)
285
286            self.set_generation_seed(self.next_seed())
287            self.generation_iteration(gen)
288            if self.should_terminate(self.population, self.best_of_run_, gen):
289                self.final_generation_ = gen
290                self.publish("after_generation")
291                break
292            self.publish("after_generation")

295        """
296        Update `gen` field for all individuals

300        gen : int
301            Current generation number
302        """
303        for subpopulation in self.population.sub_populations:
304            for ind in subpopulation.individuals:
305                ind.gen = gen

401    def should_terminate(
402        self,
403        population: Population,
404        best_of_run_: Individual,
405        generation_num: int,
406    ) -> bool:
407        if self.termination_checker is None:
408            return False
409        elif isinstance(self.termination_checker, list):
410            return any(
411                [
412                    t.should_terminate(
413                        population, best_of_run_, generation_num
414                    )
415                    for t in self.termination_checker
416                ]
417            )
418        else:
419            return self.termination_checker.should_terminate(
420                population, best_of_run_, generation_num
421            )

423    def __getstate__(self) -> Dict[str, object]:

  • random_seed (int, default=current system time): Random seed for deterministic experiment.

def evolve_main_loop(self) -> None: 
277    def evolve_main_loop(self) -> None:
278        """
279        Performs the evolutionary main loop
280        """
281        # there was already "preprocessing" generation created - gen #0
282        # now create another self.max_generation generations, starting gen #1
283        for gen in range(1, self.max_generation + 1):
284            self.generation_num = gen
285            self.update_gen(gen)
286
287            self.set_generation_seed(self.next_seed())
288            self.generation_iteration(gen)
289            if self.should_terminate(self.population, self.best_of_run_, gen):
290                self.final_generation_ = gen
291                self.publish("after_generation")
292                break
293            self.publish("after_generation")
